--Properties:

ESP.Name = "ESP"

ESP.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

Frame.Parent = ESP

Frame.BorderColor3 = Color3.fromRGB(0, 0, 0)
Frame.BorderSizePixel = 0

ESP_2.Name = "ESP"
ESP_2.Parent = Frame

ESP_2.BorderColor3 = Color3.fromRGB(0, 0, 0)
ESP_2.BorderSizePixel = 0
ESP_2.Size = UDim2.new(0, 100, 0, 100)
ESP_2.Font = Enum.Font.SourceSans
ESP_2.Text = "ESP"
ESP_2.TextColor3 = Color3.fromRGB(0, 0, 0)
ESP_2.TextSize = 14.000

-- Scripts:

local function WDBXRDK_fake_script() -- ESP_2.LocalScript 
	local script = Instance.new('LocalScript', ESP_2)

	Active = false
	
	script.Parent.MouseButton1Click:Connect(function()
		if Active == false then
			for _,v in pairs(game:GetService("Players"):GetChildren()) do 
				if v:IsA("Player") then
					char = v.Character
					Light = Instance.new("Highlight")
					Light.FillColor = Color3.fromHex("00bb2d")
					Light.Parent = char
					Active = true
				end
			end
		else
			if Active == true then
				for i,v in game:GetService("Players"):GetChildren() do
					v.Character:FindFirstChildOfClass("Highlight"):Destroy()
					Active = false
					
					
				end
			end
		end
	end)
	
	game:GetService("Players").PlayerAdded:Connect(function(player: Player) 
		if Active == true then
			char = player.Character
			Light = Instance.new("Highlight")
			Light.FillColor = Color3.fromHex("00bb2d")
			Light.Parent = char
		else
		end	
	end)
	
	game:GetService("Players").PlayerRemoving:Connect(function(player: Player) 
		if Active == true then
			player.Character:FindFirstChildOfClass("Highlight"):Destroy()
		else
		end
	end)
end
coroutine.wrap(WDBXRDK_fake_script)()
